def plot_confusion_matrix(y_true, y_pred, classes, normalize=False, title=None, cmap=plt.cm.Blues):
    if not title:
        if normalize:
            title = 'Matriz de confusión normalizada'
        else:
            title = 'Matriz de confusión sin normalizar'

    # Calcular la matriz de confusión
    cm = confusion_matrix(y_true, y_pred)
    
    # Normalizar la matriz de confusión si se solicita
    if normalize:
        cm = cm.astype('float') / cm.sum(axis=1)[:, np.newaxis]

    # Plot
    plt.figure(figsize=(8, 6))
    plt.imshow(cm, interpolation='nearest', cmap=cmap)
    plt.title(title)
    plt.colorbar()
    tick_marks = np.arange(len(classes))
    plt.xticks(tick_marks, classes, rotation=45)
    plt.yticks(tick_marks, classes)

    fmt = '.2f' if normalize else 'd'
    thresh = cm.max() / 2.
    for i in range(cm.shape[0]):
        for j in range(cm.shape[1]):
            plt.text(j, i, format(cm[i, j], fmt),
                     ha="center", va="center",
                     color="white" if cm[i, j] > thresh else "black")

    plt.ylabel('Etiqueta real')
    plt.xlabel('Etiqueta predicha')
    plt.tight_layout()
    plt.show()

def plot_ROC(y_true, y_prob):
    fpr, tpr, thresholds = roc_curve(y_true, y_prob)
    roc_auc = auc(fpr, tpr)

    plt.figure()
    plt.plot(fpr, tpr, color='darkorange', lw=2, label='ROC curve (area = %0.2f)' % roc_auc)
    plt.plot([0, 1], [0, 1], color='navy', lw=2, linestyle='--')
    plt.xlim([0.0, 1.0])
    plt.ylim([0.0, 1.05])
    plt.xlabel('Tasa de Falsos Positivos')
    plt.ylabel('Tasa de Verdaderos Positivos')
    plt.title('Curva ROC')
    plt.legend(loc="lower right")
    plt.show()

# Definir una función para entrenar y evaluar el modelo
def train_and_evaluate_model(X_train, y_train, X_test, y_test, representation):
    # Crear el clasificador Random Forest
    classifier = RandomForestClassifier(n_estimators=100, random_state=42)

    # Entrenar el modelo
    classifier.fit(X_train, y_train)

    # Predecir las etiquetas de clase en el conjunto de prueba
    y_pred = classifier.predict(X_test)

    # Calcular la matriz de confusión
    cm = confusion_matrix(y_test, y_pred)

    # Calcular precision y recall
    precision = precision_score(y_test, y_pred, pos_label='spam')
    recall = recall_score(y_test, y_pred, pos_label='spam')

    # Calcular la puntuación AUC y la curva ROC
    y_prob = classifier.predict_proba(X_test)[:, 1]

    # Convertir las etiquetas a valores binarios
    lb = LabelBinarizer()
    y_test_binary = lb.fit_transform(y_test)

    # Calcular la curva ROC y el área bajo la curva (AUC)
    fpr, tpr, _ = roc_curve(y_test_binary, y_prob)
    auc = roc_auc_score(y_test_binary, y_prob)

    # Imprimir las métricas
    print(f"Matriz de confusión ({representation}):")
    print(cm)
    print(f"Precisión ({representation}): {precision:.2f}")
    print(f"Recall ({representation}): {recall:.2f}")
    print(f"AUC ({representation}): {auc:.2f}")

    # Llamar a las funciones plot_confusion_matrix y plot_ROC
    plot_confusion_matrix(y_test, y_pred, classes=["ham", "spam"], normalize=True, title=f"Matriz de confusión ({representation})")
    plot_ROC(y_test_binary, y_prob)

2.3 Discusión¶

1. ¿Qué error es más “aceptable”: dejar pasar un SMS de SPAM (falso negativo) o bloquear un SMS legítimo (falso positivo)? Justifique su respuesta.

En el caso de la detección de spam en SMS, generalmente es más aceptable bloquear un SMS legítimo (falso positivo) que dejar pasar un SMS de spam (falso negativo). Esto se debe a que los usuarios pueden perder la confianza en el sistema si continúan recibiendo mensajes de spam, lo que puede resultar en una experiencia de usuario negativa. Por otro lado, bloquear un SMS legítimo puede ser una molestia, pero es menos perjudicial que permitir que un mensaje de spam llegue al usuario final.

2. Compare los valores para cada modelo de representación numérico. En base a la respuesta de la primera pregunta ¿Qué modelo de representación numérica produjo el mejor resultado, BoG o TF-IDF? ¿Cuál o cuáles son las razones por las que dicho modelo se comportó de mejor manera?

En base a la respuesta a la primera pregunta, el modelo que produjo el mejor resultado sería el que tiene un menor número de falsos negativos, es decir, el que tiene un mayor recall. Si observamos los resultados proporcionados, el modelo TF-IDF tiene un recall más alto en comparación con los modelos BoG (n=1) y BoG (n=2). Por lo tanto, el modelo TF-IDF sería preferible en este caso, ya que minimiza la probabilidad de dejar pasar un SMS de spam (falso negativo), lo que resulta en una mejor experiencia para el usuario final.

3. En base a la exploración de datos e ingeniería de características que realizó en el primer y este laboratorio, ¿qué consejos le daría a un familiar que le solicita ayuda para detectar si un email o SMS es phishing o no? ¿En qué características de una URL/email podría fijarse su familiar para ayudarlo a detectar un potencial phishing?

Algunos consejos para detectar si un email o SMS es phishing incluyen:

  • Verificar la autenticidad de los remitentes: Siempre revisar la dirección de correo electrónico o el número de teléfono del remitente para asegurarse de que sea legítimo.
  • No hacer clic en enlaces sospechosos: Evitar hacer clic en enlaces incluidos en correos electrónicos o mensajes de texto que parezcan sospechosos o inesperados.
  • Revisar la gramática y el formato: Prestar atención a la gramática y el formato del mensaje, ya que los correos electrónicos o mensajes de phishing suelen contener errores gramaticales o tener un formato inusual.

En cuanto a las características de una URL o email que podrían indicar un posible phishing, su familiar debería fijarse en lo siguiente:

  • El dominio de la URL: Verificar si el dominio de la URL coincide con el sitio web esperado. Los sitios web de phishing a menudo utilizan dominios falsos o similares para engañar a los usuarios.
  • Solicitudes de información confidencial: Ser cauteloso con los correos electrónicos o mensajes que solicitan información confidencial, como contraseñas, números de cuenta bancaria o números de tarjeta de crédito.
  • Urgencia o amenazas: Tener cuidado con los correos electrónicos o mensajes que inducen a la urgencia o amenazan con consecuencias negativas si no se toma una acción inmediata.

4. Si detectamos una URL o email/SMS de phishing, ¿qué podemos hacer para detener su distribución?

Algunas acciones que se pueden tomar para detener la distribución de URLs o correos electrónicos/SMS de phishing incluyen:

  • Reportar el correo electrónico o mensaje de phishing a la compañía o institución afectada para que puedan tomar medidas.
  • Bloquear la dirección de correo electrónico o número de teléfono del remitente para evitar recibir más mensajes de phishing.
  • Actualizar y mantener actualizado el software de seguridad, como los programas antivirus y antimalware, para detectar y bloquear mensajes de phishing.
  • Educar a los usuarios sobre cómo reconocer y evitar el phishing proporcionando capacitación y concienciación sobre seguridad en línea.
